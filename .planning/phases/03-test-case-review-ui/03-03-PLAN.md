```typescript
export type DocumentStatus = "pending" | "processing" | "extracting" | "done" | "error" | "partial"

}

  id: number; name: string; type: FeatureType; confidence: number
  summary: string | null; status: FeatureStatus

  overview_md: string | null

export interface DocumentResponse {
```

**9. frontend/src/pages/ProjectPage.tsx — Sidebar + Content layout:**
```typescript
import { useUIStore } from "@/stores/uiStore"
import { useDocument } from "@/hooks/useDocuments"
import { useExtractionSSE } from "@/hooks/useExtraction"
import { Sidebar } from "@/components/layout/Sidebar"
import { ContentArea } from "@/components/layout/ContentArea"

export default function ProjectPage() {
  const documentId = useUIStore((s) => s.selectedDocumentId)
  const { data: document, isLoading } = useDocument(documentId)

  // Enable SSE if document is still being processed
  const isExtracting = document?.status === "processing" || document?.status === "extracting"
  useExtractionSSE(documentId, isExtracting)

  if (isLoading || !document) return <div className="p-8">Loading...</div>

  return (
    <div className="flex h-screen">
      <Sidebar document={document} />
      <main className="flex-1 overflow-y-auto p-6">
        <ContentArea document={document} />
      </main>
    </div>
  )
}
```
